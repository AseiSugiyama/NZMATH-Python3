 \section{multiplicative -- 乗法的数論関数}\linkedzero{multiplicative}
%
このメソッドの全ての関数は,特に断りのない限り自然数のみ受け付ける.
%
  \subsection{euler -- オイラーのファイ関数}\linkedone{multiplicative}{euler}
   \func{euler}
   {%
     \hiki{n}{integer}
   }{%
     \out{integer}
   }\\
   \spacing
   % document of basic document
   \quad \param{n}と互いに素かつ\param{n}よりも小さい数の総数を返す.この関数はよく\(\varphi\) として言及される.

  \subsection{moebius -- メビウス関数}\linkedone{multiplicative}{moebius}
   \func{moebius}
   {%
     \hiki{n}{integer}
   }{%
     \out{integer}
   }\\
   \spacing
   % document of basic document
   \quad この関数は以下のいずれかの値を返す:
   \begin{description}
   \item[-1] (\param{n}が素因数に奇数をもっているとき,)
   \item[1] (\param{n}が素因数に偶数をもっているとき,)
   \item[0] (\param{n}が素因数が平方数を持っているとき.)
   \end{description}
   この関数はよく\(\mu\) として言及される.

  \subsection{sigma -- 約数の冪の合計}\linkedone{multiplicative}{sigma}
   \func{sigma}
   {%
     \hiki{m}{integer},\ %
     \hiki{n}{integer}
   }{%
     \out{integer}
   }\\
   \spacing
   % document of basic document
   \param{n}の因数の\param{m}乗を返す.
   引数\param{m}は零でもよく,因数の数を返す.
   この関数はよく\(\sigma\) として言及される.
%
\begin{ex}
>>> multiplicative.euler(1)
1
>>> multiplicative.euler(2)
1
>>> multiplicative.euler(4)
2
>>> multiplicative.euler(5)
4
>>> multiplicative.moebius(1)
1
>>> multiplicative.moebius(2)
-1
>>> multiplicative.moebius(4)
0
>>> multiplicative.moebius(6)
1
>>> multiplicative.sigma(0, 1)
1
>>> multiplicative.sigma(1, 1)
1
>>> multiplicative.sigma(0, 2)
2
>>> multiplicative.sigma(1, 3)
4
>>> multiplicative.sigma(1, 4)
7
>>> multiplicative.sigma(1, 6)
12L
>>> multiplicative.sigma(2, 7)
50
\end{ex}%Don't indent!(indent causes an error.)
