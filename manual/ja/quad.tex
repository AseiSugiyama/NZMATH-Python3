  \subsection{ReducedQuadraticForm -- 簡約二次形式クラス}\linkedone{quad}{ReducedQuadraticForm}
  \initialize
  \func{ReducedQuadraticForm}{\hiki{f}{list}, \, \hiki{unit}{list}}{\out{ReducedQuadraticForm}}\\
  \spacing
  % document of basic document
  \quad ReducedQuadraticFormオブジェクトを作成.
  \spacing
  % added document
  \spacing
  % input, output document
  \quad \param{f}, \param{unit} は \(3\) 整数 {\tt [a, b, c]}のリストでなければならず,
  二次形式を\(ax^2+bxy+cy^2\) と表記.
  \param{unit}は単元形式を表す.
  \begin{op}
    \verb|M * N| & \param{M} と \param{N}の合成を返す. \\
    \verb|M ** a| & \param{M}の$a$乗を返す. \\
    \verb|M / N| & 二次形式の除算. \\
    \verb|M == N| & \param{M} と \param{N} が等しいかどうか返す. \\
    \verb|M != N| & \param{M} と \param{N} が等しくないかどうか返す. \\    
  \end{op}

  \subsection{class\_formula}\linkedone{quad}{class\_formula}
   \func{class\_formula}{\hiki{d}{integer}, \, \hiki{uprbd}{integer}}{\out{integer}}\\
   \spacing
   % document of basic document
   \quad 類数公式を使い,判別式\param{d}を持つ類数$h$の近似値を返す. \\
   \spacing
   % added document
   \quad 類数公式 $\DS h = \frac{\sqrt{|\param{d}|}}{\pi}\prod_{p}\left(1 - \left(\frac{\param{d}}{p}\right) \frac{1}{p}\right)^{-1}$. \\
   \spacing
   % input, output document
   \quad 入力する数\param{d}はint型, long型 または \linkingone{rational}{Integer}でなければならない.  
%

(12, [ReducedQuadraticForm(1, 0, 300), ReducedQuadraticForm(3, 0, 100), 
ReducedQuadraticForm(4, 0, 75), ReducedQuadraticForm(12, 0, 25),

12L

[[2L], [2L, 0]])
