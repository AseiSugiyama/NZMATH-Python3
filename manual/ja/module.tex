\subsection{Module - 数体上の加群}\linkedone{module}{Module}
 \initialize
  \func{Module}{\hiki{pair\_mat\_repr}{list/matrix},\ \hiki{number\_field}{algfield.NumberField},\ \hikiopt{base}{list/matrix.SquareMatrix}{None},\ \hikiopt{ishnf}{bool}{False}}{\out{Module}}\\
  \spacing
  % document of basic document
  \quad 数体上の新しい加群オブジェクトを作成.\\
  \spacing
  % added document
  \quad 加群は有限生成された部分$\mathbf{Z}$加群. 
  加群の階数をdeg(number\_field) と仮定しないことを注意.\\
  加群を,$\theta$が\param{number\_field}.\linkingtwo{algfield}{BasicAlgNumber}{polynomial}の解となる$\mathbf{Z}[\theta]$上の基本的な加群についての生成元として表す.\\
  \spacing
  % input, output document
  \quad \param{pair\_mat\_repr}は次に示す形式のどれかであるべきである:
  \begin{itemize}
  \item $[M,\ d]$, $M$のサイズは\param{number\_field}の次数である整数のタプルまたはベクトルのリストであり,$d$は分母.
  \item $[M,\ d]$, $M$のサイズは\param{number\_field}の次数である整数行列,であり$d$は分母.
  \item 行の数は\param{number\_field}の次数である有理数行列.
  \end{itemize}
  また,\param{base}は次に示す形式のうちのどれかであるべきである:
  \begin{itemize}
  \item \linkedone{module}{base}サイズは\param{number\_field}の次数である有理数のタプルまたはベクトルのリスト
  \item サイズは\param{number\_field}である非特異かつ有理数係数の正方行列
  \end{itemize}
  加群は\linkingtwo{module}{Module}{base}について内部で$\frac{1}{d}M$と表され,
  $d$は\linkingtwo{module}{Module}{denominator}で$M$は\linkingtwo{module}{Module}{mat\_repr}.
  \param{ishnf}がTrueなら,\param{mat\_repr}はHNFであると仮定.\\
  \begin{at}
  \item[mat\_repr]\linkedtwo{module}{Module}{mat\_repr}:サイズが\param{number\_field}の次数である\linkingone{module}{Submodule} $M$のインスタンス 
  \item[denominator]\linkedtwo{module}{Module}{denominator}:整数 $d$
  \item[base]\linkedtwo{module}{Module}{base}:サイズは\param{number\_field}である正方かつ非特異有理数行列 
  \item[number\_field]\linkedtwo{module}{Module}{number\_field}:加群が定義された数体
  \end{at}
\begin{op}
    \verb+M==N+ & \param{M}と\param{N}が加群として等しいかどうか返す.\\
    \verb+c in M+ & \param{M}の要素のどれかが\param{c}と等しいかどうか返す.\\
    \verb|M+N| & \param{M}と\param{N}の加群としての部分集合を返す.\\
    \verb+M*N+ & \param{M}と\param{N}のイデアル計算としての積を返す. \\
               &  \param{N}は加群またはスカラーでなければならない(\linkingtwo{module}{Module}{number\_field}の要素).\\
               &  $M$と$N$の共通部分の計算したいときは\linkingtwo{module}{Module}{intersect}を参照.\\
    \verb+M**c+ & イデアルの乗算を基にした\param{M}の\param{c}乗を返す.\\
    \verb+repr(M)+ & 加群\param{M}のrepr文字列を返す.\\
    \verb+str(M)+ & 加群\param{M}のstring文字列を返す.\\
  \end{op}
\begin{ex}
>>> F = algfield.NumberField([2,0,1])
>>> M_1 = module.Module([matrix.RingMatrix(2,2,[1,0]+[0,2]), 2], F)
>>> M_2 = module.Module([matrix.RingMatrix(2,2,[2,0]+[0,5]), 3], F)
>>> print M_1
([1, 0]+[0, 2], 2)
 over
([1L, 0L]+[0L, 1L], NumberField([2, 0, 1]))
>>> print M_1 + M_2
([1L, 0L]+[0L, 2L], 6)
 over
([Rational(1, 1), Rational(0, 1)]+[Rational(0, 1), Rational(1, 1)], 
NumberField([2, 0, 1]))
>>> print M_1 * 2
([1L, 0L]+[0L, 2L], 1L)
 over
([Rational(1, 1), Rational(0, 1)]+[Rational(0, 1), Rational(1, 1)], 
NumberField([2, 0, 1]))
>>> print M_1 * M_2
([2L, 0L]+[0L, 1L], 6L)
 over
([Rational(1, 1), Rational(0, 1)]+[Rational(0, 1), Rational(1, 1)], 
NumberField([2, 0, 1]))
>>> print M_1 ** 2
([1L, 0L]+[0L, 2L], 4L)
 over
([Rational(1, 1), Rational(0, 1)]+[Rational(0, 1), Rational(1, 1)], 
NumberField([2, 0, 1]))
\end{ex}%Don't indent!

\subsection{Ideal\_with\_generator - 生成元によるイデアル}\linkedone{module}{Ideal\_with\_generator}
 \initialize
  \func{Ideal\_with\_generator}{\hiki{generator}{list}}{\out{Ideal\_with\_generator}}\\
  \spacing
  % document of basic document
  \quad 生成元により与えられた新しいイデアルを作成.\\
  \spacing
  % added document
  %\quad 
  %\spacing
  % input, output document
  \quad \param{generator}は同じ数体上の生成元を表す\linkingone{algfield}{BasicAlgNumber}のインスタンスのリスト.
  \begin{at}
  \item[generator]\linkedtwo{module}{Ideal\_with\_generator}{generator}: イデアルの生成元
  \item[number\_field]\linkedtwo{module}{Ideal\_with\_generator}{number\_field}: 生成元が定義された数体
  \end{at}
\begin{op}
    \verb+M==N+ & \param{M}と\param{N}が加群として等しいかどうか返す.\\
    \verb+c in M+ & \param{M}のどれかの要素が\param{c}と等しいかどうか返す.\\
    \verb|M+N| & \param{M}と\param{N}のイデアル生成元としての和を返す.\\
    \verb+M*N+ & \param{M}と\param{N}のイデアル生成元としての積を返す. \\
    \verb+M**c+ & イデアルの積を基にした\param{M}の\param{c}乗を返す.\\
    \verb+repr(M)+ & イデアル\param{M}のrepr文字列を返す.\\
    \verb+str(M)+ & イデアル\param{M}のstr文字列を返す.\\
  \end{op}

>>> F = algfield.NumberField([2,0,1])

[BasicAlgNumber([[1L, 0L], 3L], [2, 0, 1]), BasicAlgNumber([[0L, 5L], 6], [2, 0, 1]), 
BasicAlgNumber([[0L, 2L], 3], [2, 0, 1]), BasicAlgNumber([[-10L, 0L], 3], [2, 0, 1])]
>>> print M_1 ** 2
[BasicAlgNumber([[1L, 0L], 4], [2, 0, 1]), BasicAlgNumber([[0L, 1L], 2], [2, 0, 1]), 
BasicAlgNumber([[0L, 1L], 2], [2, 0, 1]), BasicAlgNumber([[-2L, 0L], 1], [2, 0, 1])]
