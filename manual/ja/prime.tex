 \subsection{by\_primitive\_root -- Lehmer's test}\linkedone{prime}{by\_primitive\_root}
   \func{by\_primitive\_root}
   {\hiki{n}{integer},\ \hiki{divisors}{sequence}}{\out{True/False}}\\
   \spacing
   % document of basic document
   \quad Lehmerの素数判定法~\cite{Lehmer1927}.\\
   \spacing
   % added document
   \quad \param{n} が素数のときかつそのときに限り True を返す.\\
    このメソッドは原始根の存在に基づいて \param{n} が素数であることを示す.
    このために $n - 1$ の素因数を知っていることが必要である.
   \spacing
   % input, output document
   \quad \param{divisors} は $n - 1$ の素因数のシーケンス (list, tuple, etc).
   %\quad 
%
 \subsection{full\_euler -- Brillhart \& Selfridge's test}\linkedone{prime}{full\_euler}
   \func{full\_euler}
   {\hiki{n}{integer},\ \hiki{divisors}{sequence}}{\out{True/False}}\\
   \spacing
   % document of basic document
   \quad Brillhart \& Selfridge の素数判定法~\cite{BS1967}.\\
   \spacing
   % added document
   \quad 
    このメソッドは $\varphi(n) = n - 1$ の成立により \param{n} 
    が素数であることを示す, ただし $\varphi$ はオイラーのファイ関数
    (\linkingone{multiplicative}{euler} を参照).
    このために $n - 1$ の素因数を知っていることが必要である.
   \spacing
   % input, output document
   \quad \param{divisors} は $n - 1$ の素因数のシーケンス (list, tuple, etc).
   %\quad 
   \quad

 \subsection{Lucas\_chain -- Lucas数列}\linkedone{prime}{Lucas\_chain}
   \func{Lucas\_chain}
   {\hiki{n}{integer},\ \hiki{f}{function},\ \hiki{g}{function},\ \hiki{x\_0}{integer},\ \hiki{x\_1}{integer}}{(\out{integer},\ \out{integer})}\\
   \spacing
   % document of basic document
   \quad 以下のように定義される数列$\{ x_i \}$に対する値($x_n$,\ $x_{n+1}$)を返す:\\
   \begin{eqnarray*}
      x_{2i} = \param{f}(x_i)\\
      x_{2i+1} = \param{g}(x_i, x_{i+1})\ ,
   \end{eqnarray*}
   初項は\param{x\_0},\ \param{x\_1}.\\
   \spacing
   % added document
   %\quad  
   %\spacing
   % input, output document
   \quad \param{f}は$1$変数の整数関数.
   \param{g}は$2$変数の整数関数.\\
%
\begin{ex}
>>> prime.primeq(131)
True
>>> prime.primeq(133)
False
>>> g = prime.generator()
>>> g.next()
2
>>> g.next()
3
>>> prime.prime(10)
29
>>> prime.nextPrime(100)
101
>>> prime.primitive_root(23)
5
\end{ex}%Don't indent!(indent causes an error.)
