  \subsection{gcd -- the greatest common divisor}\linkedone{gcd}{gcd}
   \func{gcd}{\hiki{a}{integer},\ \hiki{b}{integer}}{\out{integer}}\\
   \spacing
   % document of basic document
   \quad 二つの整数\param{a}と\param{b}の最大公約数の値を返す。\\
   \spacing
   % added document
   %\spacing
   % input, output document
   \quad \param{a},\ \param{b}はintまたはlong型か\linkingone{rational}{Integer}。
   Even if one of the arguments is negative, 結果は整数。\\

  \subsection{extgcd -- extended gcd algorithm}\linkedone{gcd}{extgcd}
   \func{extgcd}{\hiki{a}{integer},\ \hiki{b}{integer}}{(\out{integer},\ \out{integer},\ \out{integer})}\\
   \spacing
   % document of basic document
   \quad $d = \param{a}u + \param{b}v$の関係式を満たす\param{a}と\param{b}の最大公約数$d$の値を返す。\\
   \spacing
   % added document
   %\spacing
   % input, output document
   \quad \param{a},\ \param{b}はintまたはlong型 \linkingone{rational}{Integer}.\\
   結果は(\param{u},\ \param{v},\ \param{d})の形で返ってくる。\\

  \subsection{lcm -- the least common multiple}\linkedone{gcd}{lcm}
   \func{lcm}{\hiki{a}{integer},\ \hiki{b}{integer}}{\out{integer}}\\
   \spacing
   % document of basic document
   \quad 二つの整数\param{a}と\param{b}の最小公倍数の値を返す。\\
   \spacing
   % added document
   \negok もし\param{a}と\param{b}どちらも０ならば、enceptionを返す。\\
   \spacing
   % input, output document
   \quad \param{a},\ \param{b}はintまたはlong型、または\linkingone{rational}{Integer}.\\
%
  \subsection{gcd\_of\_list -- gcd of many integers}\linkedone{gcd}{gcd\_of\_list}
   \func{gcd\_of\_list}{\hiki{integers}{list}}{\out{list}}\\
   \spacing
   % document of basic document
   \quad 倍数の最大公約数の値を返す。\\
   \spacing
   % added document
   \quad 与えれた\param{integers} $[x_1,\ldots,x_n]$に対して、リスト$[d,\ [c_1,\ldots,c_n]]$を返す。すなわち$d=c_1 x_1+\cdots+c_n x_n$が成り立ち$d$は$x_1,\ldots, x_n$の最大公約数である。\\
   \spacing
   % input, output document
   \quad \param{integers}はintまたはlong型のリストである。\\
   この関数は$[d,\ [c_1,\ldots,c_n]]$を返す。$d$ と$\ c_i$は整数。\\
%
